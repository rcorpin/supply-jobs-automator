	// This function will update the end date box if the start date box is after the end date
	function updateEndDate(startBoxName, endBoxName) {
		var Dom = YAHOO.util.Dom;
		var dFmt = "MM/dd/yyyy";
		var startBox = Dom.get(startBoxName);
		var endBox = Dom.get(endBoxName);
		var endDt = Date.parseString(endBox.value, dFmt);
		var startDt = Date.parseString(startBox.value, dFmt);
		if(endDt != null && startDt != null && endDt.isBefore(startDt)) {
			endBox.value = startBox.value;
		}
		else if ( endDt == null && startDt != null ) {
			endBox.value = startBox.value;
		}
	};
	
	
    YAHOO.util.Event.onDOMReady(function(){

        var Event = YAHOO.util.Event,
            Dom = YAHOO.util.Dom,
            calendar;

        var dFmt = "MM/dd/yyyy";
        var showBtn = Dom.get("showstartDate");
        var startDateDisplayed = 0; // Used to track if this cal is being displayed to toggle state
        
        calendar = new YAHOO.widget.Calendar("k","divstartDate", {
            iframe:false,          // Turn iframe off, since container has iframe support.
            hide_blank_weeks:true,
            navigator:true,
            close:false
        });

        
		
        calendar.render();
		calendar.hide();

		// Assign an event handler for selecting a date on the calendar
        calendar.selectEvent.subscribe(function() {
            if (calendar.getSelectedDates().length > 0) {
                var selDate = calendar.getSelectedDates()[0];
                var dFmt = "MM/dd/yyyy";
                Dom.get("startDate").value =  selDate.format(dFmt);
                
                updateEndDate("startDate", "endDate");
                
            } else {
                Dom.get("startDate").value = "";
            }
            calendar.hide();
            startDateDisplayed = 0;
        });

        // Assign an event handler for clicking on the show calendar button
        Event.on(showBtn, "click", function() {
            if( startDateDisplayed == 0 ) {
	            var txtDate = Dom.get("startDate").value;
	            if (txtDate != '') {
					newDate = Date.parseString( txtDate, dFmt);
					if(newDate != null) calendar.select(newDate);
					else {
						// there was a problem parsing the date, reset the calendar and the text box
						calendar.reset();
						Dom.get("startDate").value = new Date().format(dFmt);
					}
	            }
	
				var seldate = calendar.getSelectedDates();
				if (seldate.length > 0) {
					// Set the pagedate to show the selected date if it exists
					calendar.cfg.setProperty("pagedate", seldate[0]);
					calendar.render();
	            }
				calendar.show();
				startDateDisplayed = 1;
            } else {
				calendar.hide();
				startDateDisplayed = 0;
            }
		});

        
        var dateBox = Dom.get("startDate");
		Event.on(dateBox, "blur", function() {
			updateEndDate("startDate", "endDate");
		});      
        
    });

<script type="text/javascript">
var searchCheckTimer;
function blockUIForSearch(hiddenTokenProperty, cookieName) {
  var token = new Date().getTime();
  $("#" + hiddenTokenProperty).val(token);
  $.blockUI({ message: $('#progressMessage') });
  searchCheckTimer = window.setInterval(function () {
    var cookieValue = $.cookie(cookieName);
    if (cookieValue == token)
  	  finishSearch();
  }, 1000);
}
	  
function finishSearch() {
 window.clearInterval(searchCheckTimer);
 $.cookie(cookieName, null);
 $.unblockUI();
}
</script>
